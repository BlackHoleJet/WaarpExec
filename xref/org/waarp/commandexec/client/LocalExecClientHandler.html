37  /**
38   * Handles a client-side channel for LocalExec

42  public class LocalExecClientHandler extends SimpleChannelUpstreamHandler {

44      /**
45       * Internal Logger
46       */
47      private static final WaarpInternalLogger logger = WaarpInternalLoggerFactory
48              .getLogger(LocalExecClientHandler.class);

51      protected LocalExecResult result;
52      protected StringBuilder back;
53      protected boolean firstMessage = true;
54      protected WaarpFuture future;
55      protected LocalExecClientPipelineFactory factory = null;
56      protected long delay;
57      protected String command;
58      protected Channel channel;
59      protected WaarpFuture ready = new WaarpFuture(true);

61       * Constructor
62       */
63      public LocalExecClientHandler(LocalExecClientPipelineFactory factory) {
64          this.factory = factory;
65      }

67      /**
68       * Initialize the client status for a new execution
69       * @param delay
70       * @param command
71       */
72      public void initExecClient(long delay, String command) {
73          this.result = new LocalExecResult(LocalExecDefaultResult.NoStatus);
74          this.back = new StringBuilder();
75          this.firstMessage = true;
76          this.future = new WaarpFuture(true);
77          this.delay = delay;
78          this.command = command;
79          // Sends the received line to the server.
80          if (channel == null) {
81          	try {
82  				ready.await();
83  			} catch (InterruptedException e) {
84  			}
85          }
86          try {
87  	        if (this.delay != 0) {
88  	        	channel.write(this.delay+" "+this.command+"\n").await();
89  	        } else {
90  	        	channel.write(this.command+"\n").await();
91  	        }
92          } catch (InterruptedException e) {
93          }
94      }

97       * @see org.jboss.netty.channel.SimpleChannelUpstreamHandler#channelConnected(org.jboss.netty.channel.ChannelHandlerContext, org.jboss.netty.channel.ChannelStateEvent)
98       */
99      @Override
100     public void channelConnected(ChannelHandlerContext ctx, ChannelStateEvent e)
101             throws Exception {
102     	channel = ctx.getChannel();
103         factory.addChannel(channel);
104         ready.setSuccess();

116     /**
117      * When closed, <br>
118      * If no messaged were received => NoMessage error is set to future<br>
119      * Else if an error was detected => Set the future to error (with or without exception)<br>
120      * Else if no error occurs => Set success to the future<br>
121      *
122      *
123      * @see org.jboss.netty.channel.SimpleChannelUpstreamHandler#channelClosed(org.jboss.netty.channel.ChannelHandlerContext, org.jboss.netty.channel.ChannelStateEvent)
124      */
125     @Override
126     public void channelClosed(ChannelHandlerContext ctx, ChannelStateEvent e)
127             throws Exception {
128         logger.debug(this.command+":"+"ChannelClosed");
129         if (!future.isDone()) {
130             // Should not be
131             finalizeMessage();
132         }
133         super.channelClosed(ctx, e);
134     }
135     /**
136      * Finalize a message
137      */
138     private void finalizeMessage() {
139         if (firstMessage) {
140             result.set(LocalExecDefaultResult.NoMessage);
141         } else {
142             result.result = back.toString();
143         }
144         if (result.status < 0) {
145             if (result.exception != null) {
146                 this.future.setFailure(result.exception);
147             } else {
148                 this.future.cancel();
149             }
150         } else {
151             this.future.setSuccess();
152         }
153     }
154     /**
155      * Waiting for the close of the exec
156      * @return The LocalExecResult
157      */
158     public LocalExecResult waitFor(long delay) {
159         if (delay <= 0) {
160             this.future.awaitUninterruptibly();
161         } else {
162             this.future.awaitUninterruptibly(delay);
163         }
164         result.isSuccess = this.future.isSuccess();
165         return result;
166     }

168     /**
169      * Action to do before close
170      */
171     public void actionBeforeClose(Channel channel) {
172     	// here nothing to do
173     }

175     @Override
176     public void messageReceived(ChannelHandlerContext ctx, MessageEvent e) {
177         // Add the line received from the server.
178         String mesg = (String) e.getMessage();
179         // If first message, then take the status and then the message
180         if (firstMessage) {
181             firstMessage = false;
182             int pos = mesg.indexOf(' ');
183             try {
184                 result.status = Integer.parseInt(mesg.substring(0, pos));
185             } catch (NumberFormatException e1) {
186                 // Error
187             	logger.debug(this.command+":"+"Bad Transmission: "+mesg+"\n\t"+back.toString());
188                 result.set(LocalExecDefaultResult.BadTransmition);
189                 back.append(mesg);
190                 actionBeforeClose(e.getChannel());
191                 WaarpSslUtility.closingSslChannel(ctx.getChannel());
192                 return;
193             }
194             mesg = mesg.substring(pos+1);
195             if (mesg.startsWith(LocalExecDefaultResult.ENDOFCOMMAND)) {
196                 logger.debug(this.command+":"+"Receive End of Command");
197                 result.result = LocalExecDefaultResult.NoMessage.result;
198                 back.append(result.result);
199                 this.finalizeMessage();
200             } else {
201             	result.result = mesg;
202             	back.append(mesg);
203             }
204         } else if (mesg.startsWith(LocalExecDefaultResult.ENDOFCOMMAND)) {
205             logger.debug(this.command+":"+"Receive End of Command");
206             this.finalizeMessage();
207         } else {
208             back.append('\n');
209             back.append(mesg);
210         }
211     }

213     @Override
214     public void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e) {
215         logger.warn(this.command+":"+"Unexpected exception from downstream while get information: "+firstMessage,
216                 e.getCause());
217         if (firstMessage) {
218             firstMessage = false;
219             result.set(LocalExecDefaultResult.BadTransmition);
220             result.exception = (Exception) e.getCause();
221             back = new StringBuilder("Error in LocalExec: ");
222             back.append(result.exception.getMessage());
223             back.append('\n');
224         } else {
225             back.append("\nERROR while receiving answer: ");
226             result.exception = (Exception) e.getCause();
227             back.append(result.exception.getMessage());
228             back.append('\n');
229         }
230         actionBeforeClose(e.getChannel());
231         WaarpSslUtility.closingSslChannel(e.getChannel());
232     }
