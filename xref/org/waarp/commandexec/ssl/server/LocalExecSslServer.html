21  package org.waarp.commandexec.ssl.server;

23  
24  import java.net.InetAddress;
25  import java.net.InetSocketAddress;
26  import java.util.concurrent.ExecutorService;
27  import java.util.concurrent.Executors;
28  import java.util.concurrent.TimeUnit;
29  
30  import org.jboss.netty.bootstrap.ServerBootstrap;
31  import org.jboss.netty.channel.socket.nio.NioServerSocketChannelFactory;
32  import org.jboss.netty.handler.execution.OrderedMemoryAwareThreadPoolExecutor;
33  import org.jboss.netty.logging.InternalLoggerFactory;
34  import org.waarp.commandexec.utils.LocalExecDefaultResult;
35  import org.waarp.common.crypto.ssl.WaarpSecureKeyStore;
36  import org.waarp.common.crypto.ssl.WaarpSslContextFactory;
37  import org.waarp.common.logging.WaarpSlf4JLoggerFactory;
38  import org.waarp.common.utility.WaarpThreadFactory;
39  
40  /**
41   * LocalExec server Main method.
42   *
43   */
44  public class LocalExecSslServer {
45  
46      static ExecutorService threadPool;
47      static ExecutorService threadPool2;
48      static OrderedMemoryAwareThreadPoolExecutor pipelineExecutor = 
49      		new OrderedMemoryAwareThreadPoolExecutor(
50  			1000, 0, 0, 200, TimeUnit.MILLISECONDS,
51  			new WaarpThreadFactory("CommandExecutor"));
52  
53      /**
54       * Takes 3 to 8 arguments (last 5 are optional arguments):<br>
55       * - mandatory arguments: filename keystorepaswwd keypassword<br>
56       * - if no more arguments are provided, it implies 127.0.0.1 + 9999 as port and no certificates<br>
57       * - optional arguments:<br>
58       *  "port"<br>
59       *  "port" "trustfilename" "trustpassword"<br>
60       *  "port" "trustfilename" "trustpassword" "addresse"<br>
61       *  "port" "trustfilename" "trustpassword" "addresse" "default delay"<br>
62       * @param args
63       * @throws Exception
64       */
65      public static void main(String[] args) throws Exception {
66          InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
67          int port = 9999;
68          InetAddress addr;
69          long delay = LocalExecDefaultResult.MAXWAITPROCESS;
70          String keyStoreFilename, keyStorePasswd, keyPassword;
71          String trustStoreFilename = null, trustStorePasswd = null;
72          byte []loop = {127,0,0,1};
73          addr = InetAddress.getByAddress(loop);
74          if (args.length >=3) {
75              keyStoreFilename = args[0];
76              keyStorePasswd = args[1];
77              keyPassword = args[2];
78              if (args.length >= 4) {
79                  port = Integer.parseInt(args[3]);
80                  if (args.length >= 6) {
81  	                trustStoreFilename = args[4];
82  	                trustStorePasswd = args[5];
83  	                if (args.length >= 7) {
84  	                    addr = InetAddress.getByName(args[6]);
85  	                    if (args.length > 7) {
86  	                        delay = Long.parseLong(args[7]);
87  	                    }
88  	                }
89                  }
90              }
91          } else {
92              System.err.println("Need at least 3 arguments: Filename KeyStorePswd KeyPswd");
93              return;
94          }
95          threadPool = Executors.newCachedThreadPool();
96          threadPool2 = Executors.newCachedThreadPool();
97          // Configure the server.
98          ServerBootstrap bootstrap = new ServerBootstrap(
99                  new NioServerSocketChannelFactory(threadPool, threadPool2));
100         // Load the KeyStore (No certificates)
101         WaarpSecureKeyStore WaarpSecureKeyStore =
102             new WaarpSecureKeyStore(keyStoreFilename, keyStorePasswd, keyPassword);
103         if (trustStoreFilename != null) {
104             // Include certificates
105             WaarpSecureKeyStore.initTrustStore(trustStoreFilename, trustStorePasswd, true);
106         } else {
107             WaarpSecureKeyStore.initEmptyTrustStore();
108         }
109         WaarpSslContextFactory waarpSslContextFactory =
110             new WaarpSslContextFactory(WaarpSecureKeyStore, true);
111         // Configure the pipeline factory.
112         bootstrap.setPipelineFactory(
113                 new LocalExecSslServerPipelineFactory(waarpSslContextFactory, delay, pipelineExecutor));
114 
115         // Bind and start to accept incoming connections only on local address.
116         bootstrap.bind(new InetSocketAddress(addr, port));
117     }
