44  /**
45   * LocalExec client.
46   *
47   * This class is an example of client.
48   *
49   * On a bi-core Centrino2 vPro: 18/s in 50 sequential, 30/s in 10 threads with 50 sequential
50   */
51  public class LocalExecClientTest extends Thread {
52  
53      static int nit = 20;
54      static int nth = 10;
55      static String command = "J:\\GG\\testexec.bat";
56      static int port = 9999;
57      static InetSocketAddress address;
58  
59      static LocalExecResult result;
60      static int ok = 0;
61      static int ko = 0;
62      static AtomicInteger atomicInteger = new AtomicInteger();
63  
64      static ExecutorService threadPool;
65      static ExecutorService threadPool2;
66      // Configure the client.
67      static ClientBootstrap bootstrap;
68      // Configure the pipeline factory.
69      static LocalExecClientPipelineFactory localExecClientPipelineFactory;
70  
71      /**
72       * Test & example main
73       * @param args ignored
74       * @throws Exception
75       */
76      public static void main(String[] args) throws Exception {
77          InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(
78                  Level.WARN));
79          InetAddress addr;
80          byte []loop = {127,0,0,1};
81          try {
82              addr = InetAddress.getByAddress(loop);
83          } catch (UnknownHostException e) {
84              return;
85          }
86          address = new InetSocketAddress(addr, port);
87          threadPool = Executors.newCachedThreadPool();
88          threadPool2 = Executors.newCachedThreadPool();
89          // Configure the client.
90          bootstrap = new ClientBootstrap(
91                  new NioClientSocketChannelFactory(threadPool, threadPool2));
92          // Configure the pipeline factory.
93          localExecClientPipelineFactory =
94                  new LocalExecClientPipelineFactory();
95          bootstrap.setPipelineFactory(localExecClientPipelineFactory);
96          try {
97              // Parse options.
98              LocalExecClientTest client = new LocalExecClientTest();
99              // run once
100             long first = System.currentTimeMillis();
101             client.connect();
102             client.runOnce();
103             client.disconnect();
104             long second = System.currentTimeMillis();
105             // print time for one exec
106             System.err.println("1=Total time in ms: "+(second-first)+" or "+(1*1000/(second-first))+" exec/s");
107             System.err.println("Result: " + ok+":"+ko);
108             ok = 0;
109             ko = 0;
110             // Now run multiple within one thread
111             first = System.currentTimeMillis();
112             for (int i = 0; i < nit; i ++) {
113                 client.connect();
114                 client.runOnce();
115                 client.disconnect();
116             }
117             second = System.currentTimeMillis();
118             // print time for one exec
119             System.err.println(nit+"=Total time in ms: "+(second-first)+" or "+(nit*1000/(second-first))+" exec/s");
120             System.err.println("Result: " + ok+":"+ko);
121             ok = 0;
122             ko = 0;
123             // Now run multiple within multiple threads
124             // Create multiple threads
125             ExecutorService executorService = Executors.newFixedThreadPool(nth);
126             first = System.currentTimeMillis();
127             // Starts all thread with a default number of execution
128             for (int i = 0; i < nth; i ++) {
129                 executorService.submit(new LocalExecClientTest());
130             }
131             Thread.sleep(500);
132             executorService.shutdown();
133             while (! executorService.awaitTermination(200, TimeUnit.MILLISECONDS)) {
134                 Thread.sleep(50);
135             }
136             second = System.currentTimeMillis();
137 
138             // print time for one exec
139             System.err.println((nit*nth)+"=Total time in ms: "+(second-first)+" or "+(nit*nth*1000/(second-first))+" exec/s");
140             System.err.println("Result: " + ok+":"+ko);
141             ok = 0;
142             ko = 0;
143 
144             // run once
145             first = System.currentTimeMillis();
146             client.connect();
147             client.runFinal();
148             client.disconnect();
149             second = System.currentTimeMillis();
150             // print time for one exec
151             System.err.println("1=Total time in ms: "+(second-first)+" or "+(1*1000/(second-first))+" exec/s");
152             System.err.println("Result: " + ok+":"+ko);
153             ok = 0;
154             ko = 0;
155         } finally {
156             // Shut down all thread pools to exit.
157             bootstrap.releaseExternalResources();
158             localExecClientPipelineFactory.releaseResources();
159         }
160     }

163      * Simple constructor

165     public LocalExecClientTest() {
166     }

168     private Channel channel;
169     /**
170      * Run method for thread
171      */
172     public void run() {
173         connect();
174         for (int i = 0; i < nit; i ++) {
175             this.runOnce();
176         }
177         disconnect();
178     }
179 
180     /**
181      * Connect to the Server
182      */
183     private void connect() {
184         // Start the connection attempt.
185         ChannelFuture future = bootstrap.connect(address);
186 
187         // Wait until the connection attempt succeeds or fails.
188         try {
189             channel = future.await().getChannel();
190         } catch (InterruptedException e) {
191         }
192         if (!future.isSuccess()) {
193             System.err.println("Client Not Connected");
194             future.getCause().printStackTrace();
195             return;
196         }
197     }
198     /**
199      * Disconnect from the server
200      */
201     private void disconnect() {
202     	// Close the connection. Make sure the close operation ends because
203         // all I/O operations are asynchronous in Netty.
204         try {
205         	Thread.sleep(10);
206             channel.close().await();
207         } catch (InterruptedException e) {
208         }
209     }
210 
211     /**
212      * Run method both for not threaded execution and threaded execution
213      */
214     public void runOnce() {
215      // Initialize the command context
216         LocalExecClientHandler clientHandler =
217             (LocalExecClientHandler) channel.getPipeline().getLast();
218         // Command to execute
219         String line = command+" "+atomicInteger.incrementAndGet();
220         clientHandler.initExecClient(0, line);
221         // Wait for the end of the exec command
222         LocalExecResult localExecResult = clientHandler.waitFor(10000);
223         int status = localExecResult.status;
224         if (status < 0) {
225             System.err.println(line+" Status: " + status + "\tResult: " +
226                     localExecResult.result);
227             ko++;
228         } else {
229             ok++;
230             result = localExecResult;
231         }
232     }
233     /**
234      * Run method for closing Server
235      */
236     private void runFinal() {
237         // Initialize the command context
238         LocalExecClientHandler clientHandler =
239             (LocalExecClientHandler) channel.getPipeline().getLast();
240         // Command to execute
241         clientHandler.initExecClient(-1000, "stop");
242         // Wait for the end of the exec command
243         LocalExecResult localExecResult = clientHandler.waitFor(10000);
244         int status = localExecResult.status;
245         if (status < 0) {
246             System.err.println("Shutdown Status: " + status + "\nResult: " +
247                     localExecResult.result);
248             ko++;
249         } else {
250             ok++;
251             result = localExecResult;
252         }
253     }
